/*Program to calculate the square root of a given number*/
/*Development-learner*/
#include <stdio.h>
#include <math.h>

int main()
{
    double number, result;
    printf("\n Enter the value of number:");
    scanf("%lf", &number);
    result=sqrt(number);
    printf("\n The square root of %.lf is %lf \n \n", number, result);
    return 0;
    
}

/*End of Program*/
